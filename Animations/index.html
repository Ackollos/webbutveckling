CSS

        
        #box {
            width: 100px;
            height: 100px;
            background-color: red;
            transition: width 2s, height 2s, background-color 2s; 
          }
            
          #box:hover {
            width: 300px;
            height: 300px;
            background-color: orange;
          }

          @media (min-width: 700px){
            #box-part{
              background-color: antiquewhite;

              margin-left: 30%;
            }

              width: 100px;

          }

HTML

        <div id="circle">Makes the circle move</div>

CSS

    @keyframes myAnimation {
        0%   {left:0px; top:0px;}
        25%  {left:350px; top:0px;}
        50%  {left:350px; top:0px;}
        75%  {left:0px; top:0px;}
        100% {left:0px; top:0px;}
    }

    #circle {
        width: 100px;
        height: 100px;
        background-color: red;
        position: absolute;
        animation-name: myAnimation;
        animation-duration: 4s;
        border-radius: 50px;
        margin-top: 10px;
        animation-delay: 1s;
        animation-iteration-count: infinite;
        margin-top: 70%;
    }

    @media (min-width: 700px){
  
        #circle-part{
            background-color: antiquewhite;
            border-radius: 15px;
            width: fit-content;
            margin-left: 35%;
            padding: 0 5% 0 5% 0;
        }

        #circle {
            width: 100px;
            height: 100px;
            background-color: red;
            margin-left: 35%;
            position: relative;
            animation-name: myAnimation;
            animation-duration: 4s;
            border-radius: 50px;
            margin-top: 10px;
            animation-delay: 1s;
            animation-iteration-count: infinite;
            display: flex;
            justify-content: center;
            align-items: center;
            text-align: center;

        @keyframes myAnimation {
            0%   {left:0px; top:0px;}
            25%  {left:400px; top:0px;}
            50%  {left:400px; top:0px;}
            75%  {left:0px; top:0px;}
            100% {left:0px; top:0px;}
        }
           
    }
Makes the circle move
